Model#guard

allows for access control

  connection: function(socket) {
    return socket.isAuthenticated;
  }
});

Model#registerActions

registers custom Client events that a Model can listen to
return Thing.registerActions({
  pivot: function(data) {
    var socket, table;
    socket = this.socket;
    table = this.table;
    return table.group(data.key).run().then(function(pivoted) {
      return socket.emit("pivot:" + data.key, pivoted);
    });
  }
});
